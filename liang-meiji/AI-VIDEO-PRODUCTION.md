### 工具栈

| 字幕生成 | Whisper / SRT生成 | $0 | 开源 |

### 自动化脚本（伪代码）
```python
from moviepy.editor import *
import edge_tts
import whisper

def generate_video(article_md, output_mp4):
    # Step 1: 提取纯文本（TTS用）
    tts_text = extract_tts_text(article_md)
    
    # Step 2: 生成音频
    audio_file = f"{article_md}.mp3"
    edge_tts.Communicate(tts_text, voice="en-US-AndrewNeural").save(audio_file)
    
    # Step 3: 根据音频时长，自动匹配图片（每5秒一张）
    audio = AudioFileClip(audio_file)
    duration = audio.duration
    
    images = load_images_from_folder("video-assets/portraits/")
    clips = []
    for i, img in enumerate(images):
        clip = ImageClip(img).set_duration(5).crossfadein(0.5)
        clips.append(clip)
        if len(clips) * 5 >= duration:
            break
    
    # Step 4: 拼接视频
    video = concatenate_videoclips(clips, method="compose")
    video = video.set_audio(audio)
    
    # Step 5: 生成字幕
    model = whisper.load_model("base")
    result = model.transcribe(audio_file)
    srt_file = generate_srt(result)
    
    # Step 6: 添加字幕
    video = add_subtitles(video, srt_file)
    
    # Step 7: 导出
    video.write_videofile(output_mp4, fps=24, codec="libx264")

# 批量处理
for article in glob("articles/*.md"):
    generate_video(article, article.replace(".md", ".mp4"))
```

### 优点

- 选择：Script to Video

- 添加背景音乐（可选）

- 选择1080p
- 点击 Generate
- 3-5分钟后下载
